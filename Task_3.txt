#include <iostream> // Підключення заголовного файлу для вводу/виводу даних

int main() {
    double x, y; // Оголошення змінних для зберігання координат точки
    
    // Виведення запиту на ввід координат точки
    std::cout << "ENTER COORDINATES (x y): ";
    
    // Читання координат точки з вводу користувача
    std::cin >> x >> y;
    
    // Перевірка, чи точка знаходиться в одній з двох заданих областей
    if ((x >= -1 && x <= 0 && y >= 0 && y <= 1) || 
        (x >= 0 && x <= 1 && y >= -1 && y <= 0)) {
        // Якщо точка знаходиться в одній з областей, вивести це повідомлення
        std::cout << "IN THE AREA!" << std::endl;
    } else {
        // Якщо точка не входить в жодну з областей, вивести це повідомлення
        std::cout << "NOT IN THE AREA!" << std::endl;
    }
    
    // Завершення програми з кодом успішного завершення (0)
    return 0;
}

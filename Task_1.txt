#include <iostream> // Підключаємо бібліотеку для вводу та виводу

int main() {
    int N; // Оголошуємо змінну для зберігання трьохзначного числа
    std::cout << "Enter a three-digit positive number: "; // Виводимо повідомлення для вводу числа
    std::cin >> N; // Зчитуємо трьохзначне число з консолі

    // Перевіряємо, чи введене число дійсно трьохзначне
    if (N < 100 || N > 999) {
        std::cout << "The number is not a three-digit number." << std::endl; // Виводимо повідомлення про помилку
        return 1; // Завершуємо програму з кодом помилки
    }

    int count = 0; // Змінна для підрахунку кількості цифр, які більші за 5

    // Отримуємо сотні
    int hundreds = N / 100; // Визначаємо сотні числа
    if (hundreds > 5) { // Перевіряємо, чи цифра більше 5
        count++; // Збільшуємо лічильник, якщо умова виконується
    }

    // Отримуємо десятки
    int tens = (N / 10) % 10; // Визначаємо десятки числа
    if (tens > 5) { // Перевіряємо, чи цифра більше 5
        count++; // Збільшуємо лічильник, якщо умова виконується
    }

    // Отримуємо одиниці
    int units = N % 10; // Визначаємо одиниці числа
    if (units > 5) { // Перевіряємо, чи цифра більше 5
        count++; // Збільшуємо лічильник, якщо умова виконується
    }

    // Виводимо результат
    std::cout << "Number of digits greater than 5: " << count << std::endl; // Виводимо кількість цифр, більших за 5

    return 0; // Завершуємо програму з кодом успіху
}
